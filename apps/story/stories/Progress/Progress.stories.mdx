import { useState, useEffect } from 'react'
import { ArgsTable, Meta, Story, Canvas } from '@storybook/addon-docs'
import Pinion from '@components/Pinion'
import Link from '@components/Link'
import Progress from '@components/Progress'
const link = {
  linkStyle: 'text',
  condition: 'internal',
  url: '/',
  hasIcon: true,
}

export const Component = (args) => {
  const interval = 25
  const [progress, setProgress] = useState(10)
  useEffect(() => {
    setTimeout(() => {
      setProgress(interval)
    }, 500)
  }, [])
  const addProgress = () => {
    if (progress + interval > 100) return
    setProgress(progress + interval)
  }
  const removeProgress = () => {
    if (progress - interval < 0) return
    setProgress(progress - interval)
  }
  return (
    <Pinion
      css={{
        d: 'grid',
        placeItems: 'center',
        m: 0,
        p: '$4',
        h: '75vh',
        bc: '$primary',
        '> div': {
          d: 'grid',
          gtc: 'repeat(2, 1fr)',
          grg: '$3',
        },
      }}
    >
      <Link
        label="Previous"
        css={{
          ff: 'row-reverse',
          c: '$white',
          svg: { transform: 'rotate(180deg)' },
        }}
        onClick={removeProgress}
        {...link}
      />
      <Link
        label="Next"
        css={{ c: '$white' }}
        onClick={addProgress}
        {...link}
      />
      <Progress progress={progress} css={{ gc: 'span 2' }} {...args} />
    </Pinion>
  )
}

# Progress Component

Click `Next` or `Previous` buttons to change the fill of the progress bar.

Accepts `progress` prop. Can pass separate `indicator: {}` css field to target progress bar styles.

Provides context for assistive technology to read the progress of the task.
